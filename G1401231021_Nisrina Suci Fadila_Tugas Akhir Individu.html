e. Imputasi Missing Value (NA)

## Tahap 1: Imputasi dengan Rata-rata PRODUK

mean_produk <- dt %>%
  group_by(Produk) %>%
  summarise(mean_harga_produk = mean(Harga, na.rm = TRUE)) %>%
  ungroup()

dt_imputasi <- dt %>%
  left_join(mean_produk, by = "Produk") %>%
  mutate(
  
    Harga_Imputasi = ifelse(is.na(Harga), mean_harga_produk, Harga)
  ) %>%
  select(-mean_harga_produk)

## Tahap 2: Imputasi dengan Rata-rata GLOBAL (Fallback)
mean_global <- mean(dt_imputasi$Harga_Imputasi, na.rm = TRUE)

dt_final <- dt_imputasi %>%
  mutate(
    Harga_Final = ifelse(is.na(Harga_Imputasi), mean_global, Harga_Imputasi)
  )

5. Metode Elbow untuk Penentuan Jumlah k menggunakan Jarak DTW

calc_within_dtw <- function(model, data){
  clust <- model@cluster
  centers <- model@centroids
  total <- 0
  
  for (k in unique(clust)) {
    members <- data[clust == k, , drop = FALSE]
    centroid <- centers[[k]]
    
    d <- sapply(1:nrow(members), function(i){
      dtw(members[i, ], centroid)$distance
    })
    
    total <- total + sum(d)
  }
  return(total)
}


# FUNGSI: Elbow Method otomatis untuk DTW K-Means
dtw_elbow <- function(data, kmax = 10){
  within <- numeric(kmax)
  
  for (k in 2:kmax){
    message("Menghitung k = ", k, " ...")
    km <- tsclust(
      series = data,
      type = "partitional",
      k = k,
      distance = "dtw_basic",
      centroid = "dba"
    )
    
    within[k] <- calc_within_dtw(km, data)
  }
  
  plot(2:kmax, within[2:kmax], type="b", pch=19,
       xlab="Jumlah Cluster",
       ylab="Total Within DTW Distance",
       main="Elbow Method for DTW K-Means")
  
  return(within)
}

Kesimpulan:

Secara keseluruhan, hasil analisis menunjukkan bahwa harga cabai merah besar di Jawa Barat cenderung bergerak serempak antar kabupaten/kota dengan pola penurunan pada triwulan II dan kenaikan kembali menjelang akhir tahun, di mana wilayah perkotaan dan beberapa kabupaten tertentu membentuk kantong harga yang relatif lebih tinggi dibanding daerah lain. Penerapan clustering time series dengan DTW dan k-means menghasilkan empat kelompok pola pergerakan harga, dengan mayoritas daerah masuk ke dalam dua cluster utama yang mencerminkan pola musiman yang mirip, sementara satu daerah muncul sebagai outlier dengan dinamika harga yang jauh berbeda. Meskipun kualitas pemisahan cluster masih terbatas (silhouette relatif rendah), hasil ini tetap memberikan gambaran bahwa struktur harga cabai merah besar di Jawa Barat kuat dipengaruhi faktor musiman, karakteristik wilayah (produksi vs konsumsi), serta kemungkinan perbedaan kondisi pasokan dan distribusi, sehingga dapat menjadi dasar awal untuk perumusan kebijakan pemantauan dan stabilisasi harga yang lebih terarah. Daerah pada cluster dengan harga tinggi dan volalitas besar perlu prioritas pengawasan pasokan dan integritas dengan kebijakan stabilisasi harga. Sedangkan patokan atau tolak ukur dapat dilihat dari daerah dengan cluster yang relatif stabil dalam perencanaan distribusi tanam bagi daerah lain.
